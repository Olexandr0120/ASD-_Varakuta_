def binary_search(arr, x):
    left, right = 0, len(arr) - 1

    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == x:
            return mid
        elif arr[mid] < x:
            left = mid + 1
        else:
            right = mid - 1

    return -1

import random
import time
import matplotlib.pyplot as plt
sizes = [100, 500, 1000, 5000, 10000, 20000]

linear_times = []
binary_times = []

for n in sizes:
    arr = sorted(random.sample(range(n * 3), n))
    x = arr[-1]  #    ( )

    start = time.time()
    linear_search(arr, x)
    linear_times.append(time.time() - start)

    start = time.time()
    binary_search(arr, x)
    binary_times.append(time.time() - start)
